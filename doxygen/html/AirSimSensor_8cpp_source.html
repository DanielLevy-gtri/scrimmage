325 
326  // If we aren't connected to AirSim, try to connect.
327  if (!client_connected_) {
328  cout << "Connecting to AirSim: ip " << airsim_ip_ << ", port " << airsim_port_ << endl;
329  sim_client_ = std::make_shared<ma::MultirotorRpcLibClient>(airsim_ip_,

332 
333  sim_client_->confirmConnection();
334 
335  // If we haven't been able to connect to AirSim, warn the user and return.
336  if (sim_client_->getConnectionState() !=
337  ma::RpcLibClientBase::ConnectionState::Connected) {
338  client_connected_ = false;
339  cout << "Warning: not connected to AirSim." << endl;
340 
341  // return std::make_shared<sc::MessageBase>();
342  return true;
343  }
344  client_connected_ = true;
345  sim_client_->enableApiControl(true);
346  }
347 
351 
352  // Setup state information for AirSim
353  sc::StatePtr &state = parent_->state_truth();
354  // convert from ENU to NED frame
355  ma::Vector3r pos(state->pos()(1), state->pos()(0), -state->pos()(2));
356 
357  enu_to_ned_yaw_.set_angle(ang::rad2deg(state->quat().yaw()));
358  double airsim_yaw_rad = ang::deg2rad(enu_to_ned_yaw_.angle());
359 
360  // pitch, roll, yaw
361  // note, the negative pitch and yaw are required because of the wsu coordinate frame
362  ma::Quaternionr qd = ma::VectorMath::toQuaternion(-state->quat().pitch(),
363  state->quat().roll(),
364  airsim_yaw_rad);
365 
366  // Send state information to AirSim
367  sim_client_->simSetVehiclePose(ma::Pose(pos, qd), true, vehicle_name_);
368 
369  // Get the camera images from the other thread
370  sc::MessagePtr<std::vector<AirSimImageType>> im_msg;
371  sc::MessagePtr<AirSimLidarType> lidar_msg;
372  bool new_lidar;
373  bool new_image;
374 
375  img_msg_mutex_.lock();
376  im_msg = img_msg_;
377  new_image = new_image_;
378  new_image_ = false;
379 
380  lidar_msg = lidar_msg_;
381  new_lidar = new_lidar_;
382  new_lidar_ = false;
383 
384  img_msg_mutex_.unlock();
385 
386  // If image is new, publish
387  if (new_image) {
388  int im_msg_size = im_msg->data.size();
389  for (int i = 0; i < im_msg_size; i++) {
390  im_msg->data[i].frame_num = airsim_frame_num_;
391  }
392 
393  if (save_airsim_data_) {
394  AirSimSensor::save_data(im_msg, state);
395  }
396 
397  img_pub_->publish(im_msg);
398  }
399 
400  // If lidar is new, publish
401  if (new_lidar) {
402  // Set frame # for all items in AirSim msg vector, assures frame number in saved images is same # as in ROS msgs
403  lidar_msg->data.frame_num = airsim_frame_num_;
404  // cout << "Lidar Frame Num" << lidar_msg->data.frame_num << endl;
405 
406  lidar_pub_->publish(lidar_msg);
407  }
408 
409  airsim_frame_num_++;
410 
411  return true;
412 }
